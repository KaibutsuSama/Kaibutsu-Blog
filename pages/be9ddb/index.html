int *ptr;

#include <iostream>
using namespace std;

int main() {
    int *ptr = new int[10];  // Allocate memory for an array of 10 integers

    // Initialize the array elements to the digits 0-9
    for (int i = 0; i < 10; i++) {
        *(ptr + i) = i;
        cout << *(ptr + i) << " ";
    }

    // Free the allocated memory
    delete[] ptr;

    return 0;
}
